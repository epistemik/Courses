Program 10: hread.c

This program shows how to open a file for reading and how to read from it character-by-character. It also shows that characters (and character constants) are really just numbers and can be manipulated as numbers. (Knowing the ASCII codes and how they're ordered can be of great help). 

#include <stdio.h>

main()
{
 FILE *infil;
 int charfromfil;

 infil = fopen("hello.txt", "r");
 if(infil == 0)
    {
     printf("Unable to open the file hello.txt\n\n");
     exit(1);
    }

 charfromfil = getc(infil);
 while(charfromfil != EOF)
    {
     if(charfromfil >= 'a' && charfromfil <= 'z')
        putchar(charfromfil - ('a' - 'A'));
     else
        putchar(charfromfil);

     charfromfil = getc(infil);
    }

 fclose(infil);
}

Program 11: hwbin.c

This program shows how to write binary numbers to a file (making it a binary file!). It also shows how to write many bytes at a time. (The display of the file hello.bin below may not look exactly the same on your screen). Please note the 'b' in the "wb" mode in the fopen statement. 

#include <stdio.h>

main()
{
 FILE *outfil;
 unsigned char spcode = 0, nlcode = 1, dotcode = 2, Hcode = 3, dcode = 4, 
               ecode = 5, lcode = 6, ocode = 7, rcode = 8, wcode = 9;
 unsigned char message[20];

 message[0] = Hcode;
 message[1] = ecode;
 message[2] = lcode;
 message[3] = lcode;
 message[4] = ocode;
 message[5] = spcode;
 message[6] = wcode;
 message[7] = ocode;
 message[8] = rcode;
 message[9] = lcode;
 message[10] = dcode;
 message[11] = dotcode;
 message[12] = nlcode;

 outfil = fopen("hello.bin", "wb");
 if(outfil == 0)
    {
     printf("Unable to open the file hello.bin\n\n");
     exit(1);
    }

 fwrite(&message[0], 1, 13, outfil);

 fclose(outfil);
}

Program 12: hrbin.c

This program shows how to read from a binary file. Please note the 'b' in the "rb" mode in the fopen statement. 

#include <stdio.h>

main()
{
 FILE *infil;
 unsigned char bytefromfil;
 unsigned char spcode = 0, nlcode = 1, dotcode = 2, Hcode = 3, dcode = 4, 
               ecode = 5, lcode = 6, ocode = 7, rcode = 8, wcode = 9;
 unsigned char message[20];

 message[nlcode] = '\n';
 message[spcode] = ' ';
 message[dotcode] = '.';
 message[Hcode] = 'H';
 message[dcode] = 'd';
 message[ecode] = 'e';
 message[lcode] = 'l';
 message[ocode] = 'o';
 message[rcode] = 'r';
 message[wcode] = 'w';

 infil = fopen("hello.bin", "rb");
 if(infil == 0)
    {
     printf("Unable to open the file hello.bin\n\n");
     exit(1);
    }

 while(fread(&bytefromfil, 1, 1, infil) == 1)
    putchar(message[bytefromfil]);

 fclose(infil);
}
