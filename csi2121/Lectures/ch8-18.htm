Question 1

factorial:
mov eax,[esp+4] ; get n
cmp eax,1 ; n<=1?
ja L1 ; no, continue
mov eax,1 ; yes, return 1
jmp exit
L1:
dec eax
push eax ; factorial n-1
call factorial ; result in eax
mov ebx,[esp+4] ; get n
mul ebx ; edx:eax = eax*ebx
exit:
ret 4 ; eax = result
